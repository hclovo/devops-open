PromptMinder has been deployed!

{{- if .Values.ingress.enabled }}

Access the application via Ingress:
{{- range $host := .Values.ingress.hosts }}
  {{- range .paths }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}{{ .path }}
  {{- end }}
{{- end }}

Make sure to add the hostname to /etc/hosts if using a local cluster:
  echo "$(minikube ip) {{ (index .Values.ingress.hosts 0).host }}" | sudo tee -a /etc/hosts

{{- else if contains "NodePort" .Values.service.type }}

Get the application URL:
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "prompt-minder-helm.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT

{{- else if contains "ClusterIP" .Values.service.type }}

Access the application via port-forward:
  export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "prompt-minder-helm.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 3000:{{ .Values.containerPort }}
  echo "Visit http://localhost:3000"

{{- end }}

Quick commands:
  kubectl get pods -l app.kubernetes.io/instance={{ .Release.Name }}
  kubectl logs -l app.kubernetes.io/instance={{ .Release.Name }} -f
